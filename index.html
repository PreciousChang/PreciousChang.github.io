–2016.10.13
–创建外键:设置了外键的列的数据是来源于主键表中数据
–如果一个列column在A表中作为主键，而A与B这两个表存在联系，通过设置B表中列column作为外键来进行联系。
–department表中有dno作为主键
–class中dno不是主键
–选择在class表中设置dno作为外键
alter table class
add constraint fk_dno foreign key(dno) references department(dno)
–给sc设置外键
–sc(sno)–> student(sno)
–sc(cno)–>course(cno)
alter table sc
add constraint fk_sno foreign key(sno) references student(sno)
alter table sc
add constraint fk_cno1 foreign key(cno) references course(cno)
–给student设置外键
–student(clno)–> class(clno)
alter table student
add constraint fk_clno foreign key(clno) references class(clno)
–给tc设置外键
–tc(tno)–> teacher(tno)
–tc(cno)–>course(cno)
alter table tc
add constraint fk_tno foreign key(tno) references teacher(tno)
alter table tc
add constraint fk_cno2 foreign key(cno) references course(cno)
–给teacher设置外键
–teacher(dno)–> department(dno)
alter table teacher
add constraint fk_dno2 foreign key(dno) references department(dno)

–唯一约束
alter table course
add constraint un_cname unique(cname)

–设置默认值
alter table student
add constraint df_ssex default(‘男’) for ssex

–创建默认值
create default df_pwd as ‘123456’
–绑定到某一列
exec sp_bindefault df_pwd,’student.pwd’

–现要求用T-SQL语句查询工厂名为“城东厂”的生产的产品信息（产品号，产品名，规格，厂名）
select dbo.product.prodNo,prodName,spec,factName
from dbo.product inner join dbo.produce
on dbo.produce.prodNo=dbo.product.prodNo
inner join dbo.factory
on dbo.factory.factNo=dbo.produce.factNo
where factName=’城西厂’
